type PaletteSetter = (v: boolean | ((o: boolean) => boolean)) => void;
let _paletteSet: PaletteSetter | null = null;
let _hotkeysBound = false;
function ensureHotkeys() {
  if (_hotkeysBound || typeof window === "undefined") return;
  _hotkeysBound = true;
  window.addEventListener("keydown", (e) => {
    if ((e.metaKey || e.ctrlKey) && e.key.toLowerCase() === "k") {
      e.preventDefault();
      _paletteSet?.((o) => !o);
    } else if (e.key === "Escape") {
      _paletteSet?.(false);
    }
  });
}

  // запуск собранного конвейера: {{input}}→запрос, шаги → sendPipeline (бэкенд протянет {{prev}})
  function runPipeline(name: string, stages: PipelineStage[], input: string) {
    const steps = materializeStages(stages, input).map((s) => ({
      label: s.label,
      model: s.model ? routeFunded(s.model, "ultra") ?? s.model : undefined,
      prompt: s.prompt,
    }));
    // оценка: ~$0.012 за шаг (грубо); гейтим лимитом трат
    gateSpend(steps.length * 0.012, `конвейер «${name}» (${steps.length} шага)`, () =>
      sendPipeline(name || "конвейер", steps, input),
    );
  }

    setChats((prev) => {
      const next = prev.map((c) => (c.id === chatId ? { ...c, folderId } : c));
      persistChats(next);
      return next;
    });
  }

  // выбор модели + цепочка запасных (если основная упадёт — берём следующую сильную).
  function resolveModel(
    image: boolean,
    agentOn: boolean,
    featureOv?: Feature,
  ): { model: string | undefined; fallbacks: string[] } {
    // Фича для резолва модели = РОВНО та, под которой пикер ХРАНИТ и ПОКАЗЫВАЕТ оверрайд
    // (composerFeature). Раньше тут пересчитывалось из agentOn (= image || renderAgent || webOn || run),
    // и при включённом веб-поиске/картинке фича разъезжалась с пикером → «выбрал модель, ушла другая».
    // composerFeature уже учитывает think-boost/картинку — отдельный флип не нужен.
    const feature = featureOv ?? composerFeature;

    if (image) {
      const ov = overrides.image;
      if (ov && ov !== "__auto__") {
        const cur = models.find((m) => m.id === ov);
        return { model: cur && isImageModel(cur) ? ov : "lustify-v8", fallbacks: [] };
      }
      return { model: autoModelId(feature, fundedModels, privatePref) ?? "lustify-v8", fallbacks: [] };
    }

    // модель процесса: ручной оверрайд по фиче иначе авто (с TEE-предпочтением); запас из топа
    let primary = modelForFeature(feature, fundedModels, overrides, privatePref);
    const isManualPick = !!overrides[feature] && overrides[feature] !== "__auto__";
    const pick = pickModel(feature, fundedModels, { preferPrivate: privatePref });

    if (isManualPick) {
      // РУЧНОЙ пик: честим как есть; уводим на лучшую (полный кошелёк) здоровую копию ТОЛЬКО если мёртв.
      const po = models.find((m) => m.id === primary);
      const selfOk = !!po && providerFunded(po.provider) && isModelHealthy(po.id);
      if (po && !selfOk) {
        const alt = modelCopies(po, models)
          .filter((m) => providerFunded(m.provider) && isModelHealthy(m.id))
          .sort((a, b) => providerWeight(b.provider) - providerWeight(a.provider))[0];
        if (alt) primary = alt.id;
      }
    } else {
      // АВТО + БАЛАНСИРОВКА: среди топ-кандидатов сравнимого качества берём провайдера с самым
      // ПОЛНЫМ кошельком → нагрузка размазывается, ни один аккаунт не сливается, пока другие полны.
      const topIds = [primary, ...pick.fallbacks.map((m) => m.id)].slice(0, 4);
      let best = primary;
      let bestW = -1;
      for (const id of topIds) {
        const mo = models.find((m) => m.id === id);
        if (!mo || !providerFunded(mo.provider) || !isModelHealthy(mo.id)) continue;
        const w = providerWeight(mo.provider);
        if (w > bestW) {
          bestW = w;
          best = id;
        }
      }
      primary = best;
    }
    // запас: СНАЧАЛА копии той же модели у оплаченных провайдеров (самый полный кошелёк первым),
    // ПОТОМ — разные сильные модели. «Пропала копия» лечится той же моделью, а не сменой модели.
    const primaryObj2 = models.find((m) => m.id === primary);
    const copyFallbacks = primaryObj2
      ? modelCopies(primaryObj2, models)
          .filter((m) => providerFunded(m.provider) && isModelHealthy(m.id))
          .sort((a, b) => providerWeight(b.provider) - providerWeight(a.provider))
          .map((m) => m.id)
      : [];
    const diverse = pick.fallbacks.map((m) => m.id);
    const chain = [...new Set([...copyFallbacks, ...diverse])].filter((id) => id !== primary);
    return { model: primary, fallbacks: chain };
  }

  type Override = Partial<{
    mode: ModeId;
    image: boolean;
    research: boolean;
    council: boolean;
    run: boolean;
    agentOn: boolean;
    deep: boolean;
    effort: EffortId;
    feature: Feature;
    // связка/мозг — две модели (агент)
    relay: boolean;
    brain: boolean;
    driver: string; // дешёвая/ведущая
    expert: string; // умная
    craftInstruction: string; // роль для связки (правка запроса)
    roleSystem: string; // роль для мозга (в system)
  }>;

  async function dispatchSend(text: string, ov?: Override) {
    const f = activeFns;
    const image = ov?.image ?? renderImage;
    const research = ov?.research ?? f.has("research");
    const council = ov?.council ?? f.has("council");
    const agentOn = ov?.agentOn ?? (image || renderAgent || !!ov?.run || webOn);
    // ось мощности (Grok): expert → думает; heavy → мозг-оркестратор; fast → быстрый путь ниже
    const eff = ov?.effort ?? effort;
    const deep = ov?.deep ?? (f.has("deep") || thinkBoost || eff === "expert");
    const effMode = ov?.mode ?? mode;

    // FAST: быстрая не-думающая модель на ЛЮБОЙ запрос (не только тривиальный), коротко, без памяти/связок.
    if (eff === "fast" && !image && !agentOn && !research && !council && !relayOn && !brainOn && !ov?.relay && !ov?.brain) {
      const chain = fastChatChain(fundedModels);
      void send(text, {
        mode: effMode,
        model: chain[0] ?? "__auto__",
        fallbacks: chain.slice(1),
        maxTokens: Math.min(BUDGET_TOK[budget], 1200),
        system: composeSystem(effMode, false, false),
        historyCap: memWindow || undefined,
        maxSpend: spendCap || undefined,
      });
      return;
    }
    // ВСПОМИНАЕМ из прошлых чатов — слой «вечной памяти».
    // local: поиск по localStorage (приватно). server: RAG-эмбеддинги на бэкенде (умнее).
    // Только для осмысленных запросов и не для генерации картинок.
    let recallText: string | undefined;
    if (!image && text && text.trim().length >= 6 && !isTrivial(text)) {
      const localFind = () => {
        const hits = searchChats(text, { excludeId: chatId, limit: 3 });
        if (hits.length) showToast(`🧠 вспомнил из ${hits.length} ${hits.length === 1 ? "чата" : "чатов"}`);
        return recallBlock(hits) || undefined;
      };
      if (memMode === "server") {
        const block = ragBlock(await ragQuery(text, 4));
        if (block) {
          showToast("🧠 вспомнил из памяти (смысловой поиск)");
          recallText = block;
        } else {
          recallText = localFind(); // сервер пуст/ошибка → мягкий фолбэк на локальный поиск
        }
      } else {
        recallText = localFind();
      }
    }
    // связка по умолчанию — из чипа, если не задано иначе и нет другого спец-пути
    // 3 чёткие функции (взаимоисключающие): «Мозг» (контролёр) приоритетнее «Улучшить промпт»
    const canTwo = !image && !agentOn && !research && !council;
    const brain = ov?.brain ?? ((brainOn || eff === "heavy") && canTwo);
    const relay = ov?.relay ?? (relayOn && !brainOn && canTwo);

    // СВЯЗКА (relay) / МОЗГ (brain) — РЕАЛЬНЫЕ пайплайны двух моделей в бэкенде:
    // дешёвая ведёт/правит → умная отвечает. Не имитация — бэкенд гоняет обе.
    if ((relay || brain) && !image) {
      const cheap = routeFunded(ov?.driver || autoModelId("uncensored", fundedModels), "uncensored") || "gemma-4-uncensored";
      const smart =
        routeFunded(ov?.expert || modelForFeature(featureForMode(effMode), fundedModels, overrides), featureForMode(effMode)) ||
        autoModelId("ultra", fundedModels) ||
        "rp:anthropic/claude-opus-4-8";
      const baseSys = composeSystem(effMode, deep, research, recallText);
      const sys = ov?.roleSystem ? [ov.roleSystem, baseSys].filter(Boolean).join("\n\n") : baseSys;
      void send(text, {
        mode: effMode,
        relay: relay || undefined,
        brain: brain || undefined,
        crafter: relay ? cheap : undefined,
        responder: relay ? smart : undefined,
        driver: brain ? cheap : undefined,
        model: smart,
        craftInstruction: relay ? (ov?.craftInstruction ?? relayCraft) : undefined,
        maxTokens: deep ? Math.min(BUDGET_TOK[budget] * 2, 8000) : BUDGET_TOK[budget],
        system: sys,
        historyCap: memWindow || undefined,
        maxSpend: spendCap || undefined,
      });
      return;
    }

    // глубоко/ультра → ВИДИМЫЙ многошаговый процесс (разбор по шагам → детальный ответ),
    // а не просто длинный текст. Это и есть отличие от чата.
    if (!image && !agentOn && !research && !council && (deep || effMode === "ultra")) {
      const planModel = routeFunded(autoModelId("ultra", fundedModels), "ultra");
      const ansModel = routeFunded(modelForFeature(featureForMode(effMode), fundedModels, overrides), featureForMode(effMode)) ?? planModel;
      void sendPipeline(
        effMode === "ultra" ? "ультра" : "глубоко",
        [
          {
            label: "Разбираю по шагам",
            model: planModel,
            fallbacks: fundedFallbacks("ultra", planModel),
            prompt: `Разбери вопрос по шагам: ключевые аспекты, что важно учесть, план ответа. Коротко, тезисами.\n\nВопрос: ${text}`,
          },
          {
            label: "Собираю детальный ответ",
            model: ansModel,
            fallbacks: fundedFallbacks(featureForMode(effMode), ansModel),
            prompt: `Опираясь на разбор ниже, дай финальный развёрнутый структурированный ответ на «${text}». Не повторяй план дословно.\n\nРазбор:\n{{prev}}`,
          },
        ],
        text,
      );
      return;
    }

    const { model, fallbacks } = resolveModel(image, agentOn, ov?.feature);
    void send(text, {
      mode: effMode,
      model,
      fallbacks,
      research,
      council,
      agent: agentOn,
      dev: agentOn && permMode !== "plan" && (permMode === "full" || devMode),
      maxTokens: deep ? Math.min(BUDGET_TOK[budget] * 2, 8000) : BUDGET_TOK[budget],
      system: composeSystem(effMode, deep, research, recallText),
      historyCap: memWindow || undefined,
        maxSpend: spendCap || undefined,
    });
  }

  // СУПЕР-ПОИСК (Ультра): фан-аут по движкам → источники → модель синтезирует с цитатами [1][2].
  async function runSuperSearch(query: string) {
    const q = query.trim();
    if (!q || busy || searchingWeb) return;
    // постоянный индикатор на всё время поиска (тост гаснет за 2.6с, а поиск до 75с — иначе кажется, что «висит»)
    const ac = new AbortController();
    searchAbortRef.current = ac;
    setSearchingWeb(true);
    let res;
    try {
      res = await superSearch(q, "deep", ac.signal);
    } finally {
      setSearchingWeb(false);
      searchAbortRef.current = null;
    }
    // юзер нажал «стоп» — не отправляем запрос дальше, просто выходим
    if (ac.signal.aborted) {
      showToast("Поиск остановлен");
      return;
    }
    if (!res.sources.length && res.error) showToast(`Поиск: ${res.error} — отвечаю как есть`);
    const model =
      routeFunded(modelForFeature(featureForMode("ultra"), fundedModels, overrides), "ultra") ?? autoModelId("ultra", fundedModels) ?? undefined;
    const sys = [sourcesPrompt(res.sources), composeSystem("ultra", false, false)]
      .filter(Boolean)
      .join("\n\n");
    void send(q, {
      mode: "ultra",
      model,
      sources: res.sources,
      system: sys,
      maxTokens: Math.min(BUDGET_TOK[budget] * 2, 8000),
    });
  }

  function executeCommand(cmd: Command, rawArg: string) {
    const arg = (rawArg ?? "").trim();

    if (cmd.kind === "prompt") {
      const inv = parseInvocation("/" + cmd.name + (arg ? " " + arg : ""));
      if (inv) dispatchSend(substitute(cmd.template, inv), { feature: cmd.feature });
      setInput("");
      return;
    }

    if (cmd.kind === "pipeline") {
      if (!arg) {
        showToast(`Добавь вопрос: /${cmd.name} …`);
        setInput("");
        return;
      }
      const inv = parseInvocation("/" + cmd.name + " " + arg);
      if (!inv) return;
      const steps = cmd.pipeline.map((s) => ({
        label: s.label ?? s.id,
        model: s.model.startsWith("auto:")
          ? autoModelId(s.model.slice(5) as Feature, fundedModels)
          : s.model,
        prompt: substitute(s.template, inv),
      }));
      void sendPipeline(cmd.name, steps, arg);
      setInput("");
      return;
    }

    const h = cmd.handler;
    if (h.startsWith("mode:")) {
      const m = h.slice(5) as ModeId;
      setMode(m);
      if (m === "image" && arg) dispatchSend(arg, { image: true });
      setInput("");
      return;
    }
    if (h.startsWith("builder:")) {
      setBuilderKind(h.slice(8) === "pipeline" ? "pipeline" : "prompt");
      setBuilderOpen(true);
      setInput("");
      return;
    }
    switch (h) {
      case "model":
      case "rate":
        setPickerOpenSignal((s) => s + 1);
        setInput("");
        break;
      case "auto":
        setOverrides({});
        saveOverrides({});
        showToast("Все процессы на авто-выбор моделей");
        setInput("");
        break;
      case "settings":
        setSettingsOpen(true);
        setInput("");
        break;
      case "budget":
        cycleBudget();
        setInput("");
        break;
      case "ghost":
        toggleGhost();
        setInput("");
        break;
      case "voice":
        toggleMic();
        setInput("");
        break;
      case "skill":
        setSkillOpen(true);
        setInput("");
        break;
      case "agentbuild":
        setAgentOpen(true);
        setInput("");
        break;
      case "relay":
        if (arg) dispatchSend(arg, { relay: true });
        else {
          setRelayOn((v) => !v);
          showToast(relayOn ? "Улучшить выкл" : "Улучшить вкл — дешёвая правит → умная отвечает");
        }
        setInput("");
        break;
      case "visibility":
        setFeaturesOpen(true);
        setInput("");
        break;
      case "memory":
        setMemoryOpen(true);
        setInput("");
        break;
      case "forget":
        memClear();
        showToast("Память чата очищена");
        setInput("");
        break;
      case "master":
        setMasterOpen(true);
        setInput("");
        break;
      case "files":
        attachRef.current?.click();
        setInput("");
        break;
      case "canvas": {
        const withArt = [...messages].reverse().find((m) => m.role === "assistant" && extractArtifact(m.content));
        const art = withArt ? extractArtifact(withArt.content) : null;
        if (art) openCanvas(art);
        else showToast("Нет кода для Канваса — попроси сгенерировать HTML/React/SVG");
        setInput("");
        break;
      }
      case "council":
        if (arg) dispatchSend(arg, { council: true });
        else setMode("ultra");
        setInput("");
        break;
      case "research":
        if (arg) dispatchSend(arg, { research: true });
        else toggleFn("research");
        setInput("");
        break;

        toggleFn("deep");
        setInput("");

        });

        break;

        else showToast("Так: /web курс биткоина");
        setInput("");
        break;
      case "supersearch":
        if (arg) void runSuperSearch(arg);
        else showToast("Так: /search кто выиграл лигу чемпионов 2026");
        setInput("");
        break;
      case "browse":
        setBrowserPanel({ url: arg || "https://google.com" });
        setInput("");
        break;
      case "agents":
        // с задачей — авто-старт; без задачи — открываем пустую панель (юзер впишет сам)

        setInput("");
        break;

        setMcpOpen(true);
        setInput("");
        break;

        setJobsOpen(true);
        setInput("");
        break;
      case "cookies":
        setCookiesOpen(true);
        setInput("");
        break;
      case "knowledge":
        setKnowledgeOpen(true);
        setInput("");
        break;
      case "pipeline":
        setPipelineSeed(arg || ""); // /pipeline <текст> → текст в {{input}} билдера
        setPipelineOpen(true);
        setInput("");
        break;
      case "uncensor":
        if (arg) {
          setInput("");
          void (async () => {
            showToast("Переписываю запрос под прямой ответ…");
            const rewritten = await uncensorRewrite(arg); // бэкенд rewrite_uncensored
            // отвечаем без цензуры на переписанный запрос (одним шагом)
            void dispatchSend(rewritten, { mode: "uncensored" });
          })();
        } else {
          showToast("Так: /uncensor <запрос> — перепишу и отвечу прямо");
          setInput("");
        }
        break;
      case "branch":
        if (messages.length) forkFrom(messages.length - 1);
        else showToast("Нечего ветвить — диалог пуст");
        setInput("");
        break;
      case "rp":
        setMode("uncensored");
        if (arg) dispatchSend(arg);
        setInput("");
        break;
      case "draw":
        if (arg) dispatchSend(arg, { image: true });
        else setMode("image");
        setInput("");
        break;
      case "run":
        if (arg) dispatchSend(arg, { run: true });
        else setMode("code");
        setInput("");
        break;
      case "persona":
        if (arg) {
          setPersona(arg);
          try {
            localStorage.setItem("taiga.persona", arg);
          } catch {
            /* ок */
          }
          showToast("Стиль сохранён — применяю ко всем ответам");
        } else {
          setPersona("");
          try {
            localStorage.removeItem("taiga.persona");
          } catch {
            /* ок */
          }
          showToast("Стиль сброшен");
        }
        setInput("");
        break;
      case "export": {
        if (!messages.length) {
          showToast("Диалог пуст");
          setInput("");
          break;
        }
        const md = messages
          .map(
            (m) =>
              `**${m.role === "user" ? "Ты" : (m.model ?? "Тайга")}:**\n\n${m.content}${m.image ? `\n\n![картинка](${m.image})` : ""}`,
          )
          .join("\n\n---\n\n");
        const blob = new Blob([`# Тайга ИИ — диалог\n\n${md}\n`], {
          type: "text/markdown;charset=utf-8",
        });
        const a = document.createElement("a");
        a.href = URL.createObjectURL(blob);
        a.download = "taiga-chat.md";
        a.click();
        URL.revokeObjectURL(a.href);
        showToast("Диалог сохранён в taiga-chat.md");
        setInput("");
        break;
      }
      case "retry": {
        const lastUser = [...messages].reverse().find((m) => m.role === "user");
        if (lastUser) dispatchSend(lastUser.content);
        else showToast("Нечего повторять");
        setInput("");
        break;
      }
      case "stop":
        stop();
        setInput("");
        break;
      case "balance":
        fetch("/api/balance")
          .then((r) => r.json())
          .then((d) => {
            if (typeof d?.usd === "number") {
              setBalance(d.usd);
              showToast(`Баланс: $${d.usd.toFixed(2)}`);
            } else showToast("Баланс недоступен");
          })
          .catch(() => showToast("Баланс недоступен"));
        setInput("");
        break;
      case "compact":
        runCompact(false);
        setInput("");
        break;
      case "clear":
        newChat();
        setInput("");
        break;
      case "help":
        setPaletteOpen(true);
        setInput("");
        break;
      case "soon":
        showToast(`«/${cmd.name}» скоро — каркас уже готов`);
        setInput("");
        break;
      default:
        setInput("");
    }
  }

  // гейт: сначала статус, потом права (deny → ask → allow), затем исполнение
  function runCommand(cmd: Command, arg: string) {
    if (cmd.status === "soon") {
      showToast(`«/${cmd.name}» скоро — каркас уже готов`);
      setInput("");
      return;
    }
    const tools = cmd.allowedTools;
    if (tools && tools.length) {
      // режим «план» — опасные инструменты (файлы/терминал/код) не выполняем
      if (permMode === "plan" && tools.some((t) => DANGEROUS_RX.test(t))) {
        showToast("Режим «план»: только смотрю и планирую. Переключи на «авто/полный» для выполнения.");
        setInput("");
        return;
      }
      // режим «полный» — не спрашиваем; иначе обычная лестница deny→ask→allow
      if (permMode !== "full") {
        const { ask, denied } = checkTools(perms, tools);
        if (denied) {
          setInput("");
          return;
        }
        if (ask) {
          // «Решает сам»: по риску + темпераменту либо решаем сами, либо спрашиваем карточкой
          const risk = toolRisk(ask);
          if (!shouldAsk(risk, temperament, 0.8)) {
            showToast(`Решил сам: выполняю «${ask}»`);
            applyVerdict(ask, cmd, arg, "allow-once");
          } else {
            setPendingPerm({ tool: ask, cmd, arg });
          }
          return;
        }
      }
    }

    if (!deep && !research && m !== "ultra") parts.push(BREVITY);
    if (deep)
      parts.push(

    return parts.filter(Boolean).join("\n\n") || undefined;

  function submit() {
    const t = input.trim();
    if (!t || busy || searchingWeb) return;
    if (t.startsWith("/")) {
      const inv = parseInvocation(t);
      const cmd = inv ? registry.get(inv.name) : undefined;
      if (cmd && inv) {
        runCommand(cmd, inv.raw);
        return;
      }
    }
    // быстрый интент: «создай агента/навык» → открыть конструктор, а не болтать
    if (/^(make|create|создай|новый|new)\s+(an?\s+)?(agent|агент[аы]?)\b/i.test(t)) {
      setAgentOpen(true);
      setInput("");
      return;
    }
    if (/^(make|create|создай|новый|new)\s+(an?\s+)?(skill|навык[аи]?)\b/i.test(t)) {
      setSkillOpen(true);
      setInput("");
      return;
    }
    setInput("");
    if (taRef.current) taRef.current.style.height = "auto";

    // супер-поиск включён → фан-аут движков + ответ с источниками (приоритет над обычной отправкой).
    // Тривиальные реплики («ок», «спасибо») не гоняем через дорогой 40-сек поиск — обычный быстрый ответ.
    if (superOn && !isTrivial(t) && !attachedImages.length && !attachedFiles.length && !activeAgent) {
      void runSuperSearch(t);
      return;
    }

    // тривиальное сообщение («ok», «спасибо») → быстрая НЕ-думающая модель, коротко.
    // Без памяти/режимов — чтобы не «думать» и не подмешивать лишнее на ерунду.
    if (isTrivial(t) && !attachedImages.length && !attachedFiles.length && !activeAgent && !relayOn && mode !== "image") {
      // Модель выбрана РУКАМИ → на «привет» отвечает ИМЕННО она (коротко), а не безымянная
      // быстрая цепочка. Иначе «выбрал Mistral, а на „hi“ ответил GPT». Авто-цепочка — только
      // когда выбор НЕ сделан. Быстрые модели остаются страховкой на случай реального отказа.
      const picked = overrides[composerFeature];
      const hasManual = !!picked && picked !== "__auto__";
      const fast = fastChatChain(fundedModels);
      let primary = hasManual ? picked : (fast[0] ?? "__auto__");
      if (!hasManual) {
        // АВТО + БАЛАНСИРОВКА: среди топ-4 быстрых берём провайдера с самым полным кошельком
        let best = primary;
        let bestW = -1;
        for (const id of fast.slice(0, 4)) {
          const mo = models.find((m) => m.id === id);
          if (!mo || !providerFunded(mo.provider) || !isModelHealthy(mo.id)) continue;
          const w = providerWeight(mo.provider);
          if (w > bestW) {
            bestW = w;
            best = id;
          }
        }
        primary = best;
      } else {
        // РУЧНОЙ: честим; копия (полный кошелёк) только если мёртв
        const po = models.find((m) => m.id === primary);
        const selfOk = !!po && providerFunded(po.provider) && isModelHealthy(po.id);
        if (po && !selfOk) {
          const alt = modelCopies(po, models)
            .filter((m) => providerFunded(m.provider) && isModelHealthy(m.id))
            .sort((a, b) => providerWeight(b.provider) - providerWeight(a.provider))[0];
          if (alt) primary = alt.id;
        }
      }
      const pObj2 = models.find((m) => m.id === primary);
      const copyFb = pObj2 ? modelCopies(pObj2, models).filter((m) => providerFunded(m.provider) && isModelHealthy(m.id)).sort((a, b) => providerWeight(b.provider) - providerWeight(a.provider)).map((m) => m.id) : [];
      const chain = [...new Set([primary, ...copyFb, ...fast])].filter(Boolean);
      void send(t, {
        mode: "chat",
        model: chain[0] ?? "__auto__",
        fallbacks: chain.slice(1),
        maxTokens: 300,
        historyCap: memWindow || undefined,
        maxSpend: spendCap || undefined,
        system: [PRODUCT_MAP, masterBlock(master), "Ответь коротко и по-человечески, 1–2 предложения, без рассуждений вслух."]
          .filter(Boolean)
          .join("\n\n"),
      });
      return;
    }

    // включён агент-связка → гоним через него (две модели)
    if (activeAgent && !attachedImages.length && !attachedFiles.length) {
      dispatchSend(t, agentOverride(activeAgent));
      return;
    }

    // вложения: картинки → зрение, доки → текст в контекст (бэкенд сам роутит)
    if (attachedImages.length || attachedFiles.length) {
      void send(t, {
        mode,
        model: "__auto__",
        images: attachedImages.length ? attachedImages : undefined,
        files: attachedFiles.length ? attachedFiles : undefined,
        system: composeSystem(mode, false, false),
      });
      setAttachedImages([]);
      setAttachedFiles([]);
      return;
    }

    // авто-роутер: по тексту сам выбираем передачу (если не зафиксировано вручную)
    if (autoRoute) {
      const intent = classifyIntent(t);
      if (intent.mode !== mode) setMode(intent.mode);
      if (intent.mode !== "chat") showToast(`Авто → ${modeRu(intent.mode)}`);
      const ov: Override =
        intent.mode === "image"
          ? { mode: "image", image: true }
          : intent.mode === "ultra"
            ? { mode: "ultra", feature: "ultra" }
            : intent.mode === "code"
              ? { mode: "code", feature: "code" }
              : intent.mode === "uncensored"
                ? { mode: "uncensored", feature: "uncensored" }
                : { mode: "chat" };
      dispatchSend(t, ov);
      return;
    }
    dispatchSend(t);
  }

  function onKey(e: React.KeyboardEvent<HTMLTextAreaElement>) {
    if (e.key === "Tab" && e.shiftKey) {
      e.preventDefault();
      cyclePermMode(); // Shift-Tab — лестница прав (как в Claude Code)
      return;
    }
    if (inlineOpen) {
      if (e.key === "ArrowDown") {
        e.preventDefault();
        setInlineActive((i) => Math.min(inlineItems.length - 1, i + 1));
        return;
      }
      if (e.key === "ArrowUp") {
        e.preventDefault();
        setInlineActive((i) => Math.max(0, i - 1));
        return;
      }
      if (e.key === "Escape") {
        e.preventDefault();
        setInput("");
        return;
      }
      if ((e.key === "Enter" && !e.shiftKey) || e.key === "Tab") {
        e.preventDefault();
        pickInline(inlineItems[Math.min(inlineActive, inlineItems.length - 1)]);
        return;
      }
    }
    if (e.key === "Enter" && !e.shiftKey) {
      e.preventDefault();
      submit();
    }
  }

  // документ → текст: читаем base64 → бэкенд /api/extract → прикрепляем как контекст
  function addDocFile(f: File) {
    setFileBusy(true);
    const fr = new FileReader();
    fr.onload = () => {
      const b64 = String(fr.result).split(",")[1] || "";
      fetch("/api/extract", {
        method: "POST",
        headers: { "content-type": "application/json" },
        body: JSON.stringify({ name: f.name, b64 }),
      })
        .then((r) => r.json())
        .then((d) => {
          if (d?.text) setAttachedFiles((prev) => [...prev, { name: f.name, text: d.text }].slice(0, 6));
          else showToast(`Не вышло прочитать ${f.name}`);
        })
        .catch(() => showToast(`Не вышло прочитать ${f.name}`))
        .finally(() => setFileBusy(false));
    };
    fr.readAsDataURL(f);
  }

// переключатель веток «‹ 2/3 ›» — появляется на сообщении, у которого есть варианты
function BranchSwitcher({
  info,
  disabled,
  onPick,
}: {
  info: { index: number; total: number; siblings: string[] };
  disabled?: boolean;
  onPick: (sibId: string) => void;
}) {
  const { index, total, siblings } = info;
  const go = (d: number) => {
    const ni = index + d;
    if (ni < 0 || ni >= total) return;
    onPick(siblings[ni]);
  };
  return (
    <div
      className="flex items-center gap-0.5 rounded-md border border-white/10 bg-white/[0.03] px-1 py-0.5 text-[11px] text-white/45"
      title="Варианты ответа — переключайся между ветками"
    >
      <button
        type="button"
        disabled={disabled || index === 0}
        onClick={() => go(-1)}
        className="px-1 leading-none transition-colors hover:text-white/80 disabled:opacity-30"
      >
        ‹
      </button>
      <span className="tabular-nums">
        {index + 1}/{total}
      </span>
      <button
        type="button"
        disabled={disabled || index === total - 1}
        onClick={() => go(1)}
        className="px-1 leading-none transition-colors hover:text-white/80 disabled:opacity-30"
      >
        ›
      </button>
    </div>
  );
}
